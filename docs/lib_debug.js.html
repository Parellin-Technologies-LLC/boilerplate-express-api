'use strict';

const
	gonfig = require( 'gonfig' );

/**
 * debug
 * @description
 * log wrapper if the API is in VERBOSE mode
 * @param {*} msg - item to log
 * @mixin debug
 */
module.exports = msg => {
	if( gonfig.log === gonfig.LEVEL.VERBOSE ) {
		console.log( msg );
	}
};
